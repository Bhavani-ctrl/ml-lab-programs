import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from sklearn.datasets import fetch_california_housing


california = fetch_california_housing(as_frame=True)
df = california.frame
print("Dataset Shape:", df.shape)
print("\nNumerical Features:")
print(df.columns)

plt.figure(figsize=(15, 12))

for i, column in enumerate(df.columns, 1):
    plt.subplot(3, 3, i)
    plt.hist(df[column], bins=30)
    plt.title(f"Histogram of {column}")
    plt.xlabel(column)
    plt.ylabel("Frequency")

plt.tight_layout()
plt.show()

plt.figure(figsize=(15, 12))

for i, column in enumerate(df.columns, 1):
    plt.subplot(3, 3, i)
    plt.boxplot(df[column], vert=False)
    plt.title(f"Box Plot of {column}")
    plt.xlabel(column)

plt.tight_layout()
plt.show()

outlier_summary = {}

for column in df.columns:
    Q1 = df[column].quantile(0.25)
    Q3 = df[column].quantile(0.75)
    IQR = Q3 - Q1
    
    lower_bound = Q1 - 1.5 * IQR
    upper_bound = Q3 + 1.5 * IQR
    
    outliers = df[(df[column] < lower_bound) | (df[column] > upper_bound)]
    outlier_summary[column] = outliers.shape[0]

print("\nOutlier Count per Feature:")
for feature, count in outlier_summary.items():
    print(f"{feature}: {count}")

import matplotlib.pyplot as plt
from collections import Counter

data = np.random.rand(100)

train = data[:50]
test = data[50:]
labels = ["Class1" if x <= 0.5 else "Class2" for x in train]

for k in [1,2,3,4,5,20,30]:

    pred = []

    for x in test:
        d = sorted((abs(x-tr), lab) for tr, lab in zip(train, labels))
        pred.append(Counter(l for _, l in d[:k]).most_common(1)[0][0])

    print(f"\nResults for k = {k}")
    for i, p in enumerate(pred, 51):
        print(f"x{i} = {test[i-51]:.4f} --> {p}")

    c1 = [test[i] for i in range(50) if pred[i] == "Class1"]
    c2 = [test[i] for i in range(50) if pred[i] == "Class2"]

    plt.figure(figsize=(10,6))
    plt.scatter(train, [0]*50,
                c=["blue" if l=="Class1" else "red" for l in labels])

    plt.scatter(c1, [1]*len(c1), c="blue", marker="x")
    plt.scatter(c2, [1]*len(c2), c="red", marker="x")

    plt.title(f"k-NN Classification Results for k = {k}")
    plt.grid()

print("\nClassification Complete!")

import matplotlib.pyplot as plt

np.random.seed(42)

X = np.linspace(0, 2*np.pi, 100)
y = np.sin(X) + 0.1*np.random.randn(100)

tau = 0.5
y_pred = []

for x in X:
    w = np.exp(-(X-x)**2/(2*tau**2))
    y_pred.append(np.sum(w*y)/np.sum(w))

plt.scatter(X, y, color='red', label='Data')
plt.plot(X, y_pred, color='blue', label='LWR Fit')
plt.xlabel('X')
plt.ylabel('Y')
plt.title('Locally Weighted Regression')
plt.legend()
plt.grid()
